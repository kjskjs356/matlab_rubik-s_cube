```matlab
%값 1= 위 시계 		U
%값 2= 위 반시계		U'
%값 3= 아래 시계		D
%값 4= 아래 반시계	D'
%값 5= 왼쪽 시계		L
%값 6= 왼쪽 반시계	L'
%값 7= 오른쪽 시계	R
%값 8= 오른쪽 반시계	R'
%값 9= 앞 시계		F
%값 10= 앞 반시계		F'
%값 11= 뒤 시계		B
%값 12= 뒤 반시계		B'
```
